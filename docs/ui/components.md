# UI Component Specifications

> Detailed specifications for UI components that don't exist in v1 or need significant changes for v2.

## Component Library

Omni v2 UI uses [shadcn/ui](https://ui.shadcn.com/) components with Tailwind CSS. All custom components should follow shadcn patterns.

## API Integration Patterns

### React Query Setup

```typescript
// packages/ui/src/lib/query-client.ts

export const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 30_000,        // 30 seconds
      refetchOnWindowFocus: false,
      retry: 1,
    },
    mutations: {
      onError: (error) => {
        toast.error("Operation failed", {
          description: error.message,
        });
      },
    },
  },
});
```

### Optimistic Updates

```typescript
const deleteMutation = useMutation({
  mutationFn: (id: string) => api.providers.delete(id),
  onMutate: async (id) => {
    // Cancel outgoing refetches
    await queryClient.cancelQueries({ queryKey: ['providers'] });

    // Snapshot previous value
    const previous = queryClient.getQueryData(['providers']);

    // Optimistically update
    queryClient.setQueryData(['providers'], (old: Provider[]) =>
      old.filter((p) => p.id !== id)
    );

    return { previous };
  },
  onError: (err, id, context) => {
    // Rollback on error
    queryClient.setQueryData(['providers'], context?.previous);
  },
  onSettled: () => {
    // Refetch after mutation
    queryClient.invalidateQueries({ queryKey: ['providers'] });
  },
});
```

### Real-time Updates via WebSocket

```typescript
// packages/ui/src/hooks/useWebSocketUpdates.ts

export function useWebSocketUpdates<T>(
  endpoint: string,
  queryKey: string[],
  transform?: (data: unknown) => T
) {
  const queryClient = useQueryClient();

  useEffect(() => {
    const ws = new WebSocket(`${getWsUrl()}${endpoint}`);

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      const transformed = transform ? transform(data) : data;

      // Update query cache
      queryClient.setQueryData(queryKey, (old: T[] | undefined) => {
        if (!old) return [transformed];
        // Merge or update based on your needs
        return [...old, transformed];
      });
    };

    return () => ws.close();
  }, [endpoint, queryKey, transform, queryClient]);
}
```
